cmake_minimum_required(VERSION 3.16)
project(Qt6QmlModuleApp LANGUAGES CXX)

set(CMAKE_PREFIX_PATH "/opt/Qt/6.9.0/gcc_64") # ✅ 用户指定 Qt 安装路径
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_AUTOMOC ON)

find_package(Qt6 REQUIRED COMPONENTS Core Gui Qml Quick QuickControls2)

qt_add_executable(Qt6QmlModuleApp
    main.cpp
    TerminalBackend.cpp
    TerminalBackend.h
)

qt_add_qml_module(Qt6QmlModuleApp
    URI MyApp
    VERSION 1.0
    QML_FILES qml/Main.qml
)

target_link_libraries(Qt6QmlModuleApp
    PRIVATE
    Qt6::Core
    Qt6::Gui
    Qt6::Qml
    Qt6::Quick
    Qt6::QuickControls2
)
